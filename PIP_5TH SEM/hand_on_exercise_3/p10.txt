mark = (float)(input("Enter the marks\n"))
if mark>=90:
    print("O");
elif mark>=80 and mark<90:
    print("A");
elif mark>=70 and mark<80:
    print("B");
elif mark>=60 and mark<70:
    print("C");
elif mark>=50 and mark<60:
    print("D");
elif mark>=40 and mark<50:
    print("E");
else:
    print("F");
